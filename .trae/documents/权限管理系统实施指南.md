# 权限管理系统实施指南

## 1. 实施概述

本指南详细说明如何在现有比赛管理系统中实施三级角色权限控制（Admin、Organizer、Judge），包括数据库改造、API权限控制、前端界面适配和安全策略实施。

实施将确保不同角色用户只能访问其职责范围内的功能和数据，提升系统安全性和用户体验。

```sql
CREATE TABLE judge_assignments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL REFERENCES tenants(id),
    judge_id UUID NOT NULL REFERENCES users(id),
```

```sql
-- 添加创建者字段
ALTER TABLE competitions ADD COLUMN IF NOT EXISTS creator_id UUID REFERENCES users(id);

-- 更新现有数据（设置为第一个admin用户）
```

### 3.4 行级安全策略

```sql
-- 启用行级安全
ALTER TABLE judge_assignments ENABLE ROW LEVEL SECURITY;

-- 租户隔离策略
CREATE POLICY "Judge assignments tenant isolation" ON judge_assignments
    FOR ALL USING (tenant_id = current_setting('app.current_tenant_id')::UUID);

-- 角色权限策略
CREATE POLICY "Organizers manage their competitions" ON competitions
    FOR ALL USING (
        tenant_id = current_setting('app.current_tenant_id')::UUID AND
        (current_setting('app.current_user_role') = 'admin' OR 
         creator_id = current_setting('app.current_user_id')::UUID)
    );

CREATE POLICY "Judges view assigned competitions" ON judge_assignments
    FOR SELECT USING (
        tenant_id = current_setting('app.current_tenant_id')::UUID AND
        (current_setting('app.current_user_role') = 'admin' OR 
         judge_id = current_setting('app.current_user_id')::UUID)
    );
```

## 4. 后端权限控制实施

### 4.1 权限验证中间件

```typescript
// src/middleware/permission.ts
import { NextRequest, NextResponse } from 'next/server'
import { getToken } from 'next-auth/jwt'

export interface Permission {
  resource: string
  action: string
  conditions?: Record<string, any>
}

export interface UserContext {
  id: string
  role: 'admin' | 'organizer' | 'judge'
  tenantId: string
  permissions: string[]
}

export class PermissionChecker {
  static checkPermission(user: UserContext, permission: Permission): boolean {
    // Admin拥有所有权限
    if (user.role === 'admin') {
      return true
    }

    // 检查具体权限
    const permissionKey = `${permission.resource}:${permission.action}`
    
    if (user.role === 'organizer') {
      return this.checkOrganizerPermission(user, permission)
    }
    
    if (user.role === 'judge') {
      return this.checkJudgePermission(user, permission)
    }
    
    return false
  }

  private static checkOrganizerPermission(user: UserContext, permission: Permission): boolean {
    const allowedPermissions = [
      'competition:create',
      'competition:read',
      'competition:update',
      'competition:delete',
      'participant:create',
      'participant:read',
      'participant:update',
      'participant:delete',
      'program:create',
      'program:read',
      'program:update',
      'program:delete',
      'judge:assign',
      'competition:control',
      'data:export'
    ]
    
    const permissionKey = `${permission.resource}:${permission.action}`
    return allowedPermissions.includes(permissionKey)
  }

  private static checkJudgePermission(user: UserContext, permission: Permission): boolean {
    const allowedPermissions = [
      'competition:read',
      'program:read',
      'score:create',
      'score:read',
      'score:update'
    ]
    
    const permissionKey = `${permission.resource}:${permission.action}`
    return allowedPermissions.includes(permissionKey)
  }
}

export async function withPermission(
  request: NextRequest,
  permission: Permission,
  handler: (user: UserContext) => Promise<NextResponse>
): Promise<NextResponse> {
  try {
    const token = await getToken({ req: request })
    
    if (!token) {
      return NextResponse.json({ error: '未授权访问' }, { status: 401 })
    }

    const user: UserContext = {
      id: token.sub as string,
      role: token.role as any,
      tenantId: token.tenantId as string,
      permissions: token.permissions as string[] || []
    }

    if (!PermissionChecker.checkPermission(user, permission)) {
      return NextResponse.json({ error: '权限不足' }, { status: 403 })
    }

    return await handler(user)
  } catch (error) {
    return NextResponse.json({ error: '权限验证失败' }, { status: 500 })
  }
}
```

### 4.2 API接口改造示例

```typescript
// src/app/api/competitions/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { withPermission, UserContext } from '@/middleware/permission'
import { prisma } from '@/lib/prisma'

export async function GET(request: NextRequest) {
  return withPermission(
    request,
    { resource: 'competition', action: 'read' },
    async (user: UserContext) => {
      let whereClause: any = {
        tenantId: user.tenantId
      }

      // Organizer只能查看自己创建的比赛
      if (user.role === 'organizer') {
        whereClause.creatorId = user.id
      }
      
      // Judge只能查看被分配的比赛
      if (user.role === 'judge') {
        const assignments = await prisma.judgeAssignment.findMany({
          where: {
            judgeId: user.id,
            isActive: true
          },
          select: { competitionId: true }
        })
        
        whereClause.id = {
          in: assignments.map(a => a.competitionId)
        }
      }

      const competitions = await prisma.competition.findMany({
        where: whereClause,
        include: {
          participants: true,
          programs: true
        }
      })

      return NextResponse.json(competitions)
    }
  )
}

export async function POST(request: NextRequest) {
  return withPermission(
    request,
    { resource: 'competition', action: 'create' },
    async (user: UserContext) => {
      const body = await request.json()
      
      const competition = await prisma.competition.create({
        data: {
          ...body,
          tenantId: user.tenantId,
          creatorId: user.id
        }
      })

      return NextResponse.json(competition)
    }
  )
}
```

### 4.3 权限管理API

```typescript
// src/app/api/permissions/check/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { getToken } from 'next-auth/jwt'
import { PermissionChecker, UserContext } from '@/middleware/permission'

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url)
    const resource = searchParams.get('resource')
    const action = searchParams.get('action')
    
    if (!resource || !action) {
      return NextResponse.json(
        { error: '缺少必要参数' },
        { status: 400 }
      )
    }

    const token = await getToken({ req: request })
    if (!token) {
      return NextResponse.json(
        { allowed: false, reason: '未登录' },
        { status: 401 }
      )
    }

    const user: UserContext = {
      id: token.sub as string,
      role: token.role as any,
      tenantId: token.tenantId as string,
      permissions: token.permissions as string[] || []
    }

    const allowed = PermissionChecker.checkPermission(user, {
      resource,
      action
    })

    return NextResponse.json({
      allowed,
      reason: allowed ? null : '权限不足'
    })
  } catch (error) {
    return NextResponse.json(
      { error: '权限检查失败' },
      { status: 500 }
    )
  }
}
```

## 5. 前端界面适配实施

### 5.1 权限控制组件

```typescript
// src/components/PermissionGuard.tsx
import { useSession } from 'next-auth/react'
import { ReactNode } from 'react'

interface PermissionGuardProps {
  resource: string
  action: string
  children: ReactNode
  fallback?: ReactNode
}

export function PermissionGuard({
  resource,
  action,
  children,
  fallback = null
}: PermissionGuardProps) {
  const { data: session } = useSession()
  
  if (!session?.user) {
    return <>{fallback}</>
  }

  const user = session.user
  
  // Admin拥有所有权限
  if (user.role === 'admin') {
    return <>{children}</>
  }

  // 检查具体权限
  const hasPermission = checkUserPermission(user, resource, action)
  
  return hasPermission ? <>{children}</> : <>{fallback}</>
}

function checkUserPermission(user: any, resource: string, action: string): boolean {
  const permissionKey = `${resource}:${action}`
  
  if (user.role === 'organizer') {
    const organizerPermissions = [
      'competition:create',
      'competition:read',
      'competition:update',
      'competition:delete',
      'participant:create',
      'participant:read',
      'participant:update',
      'judge:assign',
      'data:export'
    ]
    return organizerPermissions.includes(permissionKey)
  }
  
  if (user.role === 'judge') {
    const judgePermissions = [
      'competition:read',
      'program:read',
      'score:create',
      'score:read',
      'score:update'
    ]
    return judgePermissions.includes(permissionKey)
  }
  
  return false
}
```

### 6.1 权限测试用例

```typescript
// tests/permissions.test.ts
import { PermissionChecker, UserContext } from '@/middleware/permission'

describe('权限控制测试', () => {
  const adminUser: UserContext = {
    id: 'admin-1',
    role: 'admin',
    tenantId: 'tenant-1',
    permissions: []
  }
  
  const organizerUser: UserContext = {
    id: 'organizer-1',
    role: 'organizer',
    tenantId: 'tenant-1',
    permissions: []
  }
  
  const judgeUser: UserContext = {
    id: 'judge-1',
    role: 'judge',
    tenantId: 'tenant-1',
    permissions: []
  }

  test('Admin应该拥有所有权限', () => {
    expect(PermissionChecker.checkPermission(adminUser, {
      resource: 'user',
      action: 'delete'
    })).toBe(true)
    
    expect(PermissionChecker.checkPermission(adminUser, {
      resource: 'system',
      action: 'configure'
    })).toBe(true)
  })
  
  test('Organizer应该能管理比赛', () => {
    expect(PermissionChecker.checkPermission(organizerUser, {
      resource: 'competition',
      action: 'create'
    })).toBe(true)
    
    expect(PermissionChecker.checkPermission(organizerUser, {
      resource: 'judge',
      action: 'assign'
    })).toBe(true)
  })
  
  test('Organizer不应该能管理用户', () => {
    expect(PermissionChecker.checkPermission(organizerUser, {
      resource: 'user',
      action: 'create'
    })).toBe(false)
  })
  
  test('Judge应该能评分', () => {
    expect(PermissionChecker.checkPermission(judgeUser, {
      resource: 'score',
      action: 'create'
    })).toBe(true)
  })
  
  test('Judge不应该能创建比赛', () => {
    expect(PermissionChecker.checkPermission(judgeUser, {
      resource: 'competition',
      action: 'create'
    })).toBe(false)
  })
})
```
